function Promise(executor) {
    let self = this
    self.status = 'pending' //promise1的状态
    self.data ='' //promise1的值
    self.reason = ''
    self.onResolvedCallbacks = []
    self.onRejectedCallbacks = []
    //promise在pending状态可以变成fulfilled或rejected状态，状态不可逆
    function resolve(value) { 
        //onFulfilled和onRejected只允许在 execution context 栈仅包含平台代码时允许
        //这里的平台代码指的是引擎、环境以及promise实施代码。事件中要确保onFulfilled和onRejected异步执行（微任务），且应该在then方法调用的那一轮事件循环之后新执行栈中执行。
        setTimeout(function() {
            if(self.status === 'pending') {
                self.status = 'fulfilled'
                self.data = value 
                //执行resolve回调函数，将value传递到callback中
                //2.2.6.1 如果promise变成fulfilled状态，then方法可以被多次调用，所有的onFulfilled回调都需要按照then的顺序执行
                self.onResolvedCallbacks.forEach(cb => cb(value))
            }
        }, 0)
    }
    function reject(reason) {
      //onFulfilled和onRejected只允许在 execution context 栈仅包含平台代码时允许
      //这里的平台代码指的是引擎、环境以及promise实施代码。事件中要确保onFulfilled和onRejected异步执行（微任务），且应该在then方法调用的那一轮事件循环之后新执行栈中执行。
      setTimeout(function() {
        if(self.status === 'pending') {
            self.status = 'rejected'
            self.reason = reason
            //执行reject回调函数，将reason传递到callback中
            //then方法被调用时如果promise1是pending状态，则注册onResolved/onRejected回调到onResolvedCallbacks中
            //2.2.6.2 如果promise变成rejected状态，then方法可以被多次调用，所有的onRejected回调都需要按照then的顺序执行
            self.onRejectedCallbacks.forEach(cb => cb(reason))
        }
      }, 0)
    }

    try {
        executor(resolve,reject)
    } catch(e) {
        reject(e)
    }
}
/** 
    实现then链式调用有两个要求：
        1. 在实例方法then后面可以直接使用实例方法then
        2. 在前面的一个实例方法then返回一个值，不管是什么值，在后面调用的then方法中都可以获取到。
    即我们可以在实例方法then返回一个新的Promise对象，把实例方法then返回的值value，通过resolve(value)或reject(value)传递出去。
    在这期间，要保证then返回的值是状态不为pending的最终值或普通对象。
*/
Promise.prototype.then = function(onFulfilled, onRejected) {
    let promise2
    let self = this
    onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : value => value
    onRejected = typeof onRejected === 'function' ? onRejected : reason => {throw(reason)}
    let promise2 = new Promise((resolve, reject) => {
      //因为then可以多次调用，所以当不是第一次调用then的时候，promise的状态可能是fulfilled或rejected，所以分情况讨论
        if(self.status === 'fulfilled') { 
            setTimeout(function() {
                try { //try里面代码需异步进行，即异步进行resolved/rejected 回调
                    let x = onFulfilled(self.data) 
                    resolvePromise(promise2, x, resolve, reject)
                }catch (e) {
                    reject(e)
                }
            },0)
        }else if(self.status === 'rejected') {
            setTimeout(function() {
                try { //try里面代码需异步进行，即异步进行resolved/rejected 回调
                    let x = onRejected(self.reason)
                    resolvePromise(promise2, x, resolve, reject)
                } catch (e) {
                    reject(e)
                }
            },0)
            //保证then方法执行的时机在executor构造函数的异步操作执行完之后执行
            //then实例方法中，当Promise2的状态为pending的时候，先将陈成功的回调onFulfilled和失败的回调onRejeed分别存放起来（onResolvedCallbacks/onRejectedCallbacks).等executor函数中异步任务执行完毕，触发内置的resolve/reject，再其中依次执行这些回调函数
        } else if(self.status === 'pending') {
            self.onResolvedCallbacks.push(() => {
                try {
                    let x = onFulfilled(self.data)
                    resolvePromise(promise2, x, resolve, reject)
                } catch (e) {
                    reject(e)
                }
            })
            self.onRejectedCallbacks(() => {
                try {
                    let x = onRejected(self.reason)
                    resolvePromise(promise2, x, resolve, reject)
                } catch (e) {
                    reject(e)
                }
            })
        }
    })
    return promise2 //链式调用
   
}
/*
    resolvePromise 函数即根据x的值(不同类型的返回值)来决定promise2的状态的函数
    x为 `promise2 = promise1.then(onResolved, onRejected)`中onResolved/onRejected的返回值
    传入的resolve/reject 即为Promise2的executor两个实参
*/
/** 
    判断typeof then === 'function'时其实是在判断返回 的x是不是一个promise。如果没有then函数，那么x是一个普通值，直接返回resolve(x)。
        如果有then函数，那么x是一个Promise，此时需要递归解析这个promise，知道其返回值是一个普通值后作为最后的结果返回

*/
function resolvePromise(promise2, x, resolve, reject) {
    let thenCalledOrThrowed = false
    if(promise2 === x) { //当Promise2与x相等的时候，可能会存在循环引用的问题
        /** 
         * x===promise2,相当于promise2.then之后return了自己,因为then会等待return后的promise，导致一直等待 
         * let p = new Promise((resolve, reject) => {
                setTimeout(()=>{reject(1.1)},1000)
            })
            let pthen = p.then(value => {
                return pthen
            }, err=>{
                return pthen 
            })
         * */       
        reject(new TypeError('chaining cycle detected for promise')) 
        return
    }
    if(x instanceof Promise) {
        if(x.status === 'pending') {
            //若x的状态还没有确定，那么它可能被一个thenable决定最终的状态和值的
            //所以需要做一下处理，而不能认为它被一个正常的“值”resolve
            x.then(value => {
                resolvePromise(promise2, value, resolve, reject) 
            }, reason => {
                reject(reason)
            })
        } else {
            x.then(resolve, reject)
        }
        return //不进入下一个if
    }
    if(x !== null && (typeof x === 'function' || typeof x === 'object')) {
        try{
            let then = x.then // 2.3.3.1 because x.then could be a getter
            if(typeof then === 'function') {
                //x是一个promise
                then.call(x, value => {
                    if(thenCalledOrThrowed) return //2.3.3.3.3
                    thenCalledOrThrowed = true
                    resolvePromise(promise2, value, resolve, reject)
                    return
                }, reason => {
                    if(thenCalledOrThrowed) return //2.3.3.3.3
                    thenCalledOrThrowed = true
                    reject(reason)
                    return
                })
            } else {
                //x仅仅是一个函数或普通对象
                resolve(x)
            }
        } catch(e) {
            if(thenCalledOrThrowed) return //2.3.3.3.4.1
            thenCalledOrThrowed = true
            reject(e) //2.3.3.3.4.3
        }
    } else {
        //x返回基本类型，直接resolve
        resolve(x)
    }

}

# 实例方法catch实现

this.catch 就是 this.then(null, onRejected) 的别名 ，即：

	catch(onRejected) {    
		this.then(null, onRejected)  
	}

# 静态方法Promise.resolve()实现

Promise.resolve()用法: 把传入的参数转为一个Promise对象
1. 如果参数是一个Promise实例，直接返回这个实例
2. 如果参数是一个thenable对象（具有then方法的对象），Promise.resolve()会将这个对象转为Promise对象，然后立即执行then方法
3. 参数是不具有then方法的对象或根本不是对象，返回一个新的Promise实例，状态为已成功，并把参数传递出去
4. 不具有任何参数，直接返回新的Promise实例，状态为已成功。

# 静态方法 Promise.reject()实现

Promise.reject()用法： 返回一个新的Promise实例，状态为已失败，并把参数作为失败的原因传递出去。

	class Promise  {  
		//...  
		static reject(param) {  
			return new Promise((resolve, reject) => {  
				reject(param)  
			})  
		}  
	}

# 静态方法 Promise.all()实现

Promise.all用法：把多个Promise实例包装成一个新的Promise实例（p）
1. 多个Promise实例的状态均为已成功，返回的p状态为已成功
2. 其中一个Promise实例的状态为失败，则返回的p状态为已失败

	class Promise {
		//...
		static all(promises) {
			//将参数promises转为一个真正的数组
			promises = Array.from(promises)
			return new Promise((resolve, reject) => {
				const length = promises.length
				let value = [] //存放promises的返回值，数组

				if(length) { //value的个数为length
					value = Array.apply(null, {
						length: length
					})
					for(let i = 0; i < length; i++) {
						Promise.resolve(promises[i]).then(res => {
							value[i] = res
							if(i === length) {
								resolve(value)
							}
						}, err => {
							reject(err)
							return
						})
					}
				} else {
					resolve(value) //返回空数组
				}
			})
		}
	}

# 静态方法 Promise.race()实现

Promise.race用法：把多个Promise实例包装成一个新的Promise实例（p）
1. 多个Promise实例的状态均为已失败，返回的p状态为已失败
2. 其中一个Promise实例的状态改变，则返回的p状态随之改变，状态为改变的promises实例
3. promises使用Promise.resolve 需先转为promise实例

    class Promise {
        //...
        static race(promises) {
            promises = Array.from(promises) //转为数组
            return new Promise((resolve,reject) => {
                const length = promises.length
                for(let i = 0; i < length; i++) {
                    Promise.resolve(promises).then(res => {
                        resolve(res)
                        return
                    }, err => {
                        reject(err)
                        return
                    })
                }
            })
        }
    }
