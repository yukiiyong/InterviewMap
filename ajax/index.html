# Ajax是什么

Ajax 的全称是 asyncchronous JavaScript and Xml,即异步JavaScript和Xml。它是一种异步加载数据的机制，可以使得我们再不刷新整个页面的情况下请求数据，实现页面的局部刷新。

优点：实现异步通信，速度快，局部刷新页面，体验好。Ajax出现之前一直都是服务端渲染。即服务器去处理页面数据并填充到页面上，然后返回整个页面给客户端。

# Ajax 相关参数

  1. responseXML 接收响应的字符串类型数据
  2. responseText 接收"text/xml","application/xml"格式响应
  3. status 响应的HTTP状态码
  4. readyState 请求和响应的当前阶段
  5. timeout 超时时间

# readyState值以及意义

  1. xhr.readyStatus==0 尚未调用open方法
  2. xhr.readyStatus==1 已调用open方法但尚未发送请求（未调用send方法）
  3. xhr.readyStatus==2 已发生请求（已调用send方法），但未收到响应
  4. xhr.readyStatus==3 已接收到请求返回的数据
  5. xhr.readyStatus==4 请求已完成

# promise实现Ajax

  1. 无论是get或post，传入的参数必须经过字符串拼接后编码(encodeURIComponent)
  2. open方法用于初始化一个请求，传入method(get/post)、url、async
  3. post请求在send之前要setRequestHeader('Content-Type','application/x-www-form-urlencoded;charset=UTF-8')
function customAjax(options) {
  let url = options.url
  const method = options.method.toLocaleLowerCase() || 'get'
  const async = options.async !== false //default is true
  const data = options.data
  const xhr = new XMLHttpRequest()

  if(options.timeout && options.timeout > 0) {
    xhr.timeout = options.timeout //网络不佳时，为请求设置一个超时时间
  }

  return new Promise((resolve, reject) => {
    xhr.ontimeout = () => reject && reject('请求超时')
    xhr.onreadystatechange = () => {
      if(xhr.readyState === 4) {
        //HTTP 状态在200-300 表示请求成功
        //HTTP 状态为 304 表示请求内容未改变，可以直接从缓存中读取
        if(xhr.status === 200 && xhr.status < 300 || xhr.status == 304) {
          resolve && resolve(xhr.responseText)
        } else {
          reject && reject()
        }
      }
    }
    xhr.onerror = (e) => reject && reject(e) 
    //组建参数
    let paramArr = []
    let encodeData
    //把data序列化 
    if(data instanceof Object) {
      for(let key in data) {
        paramArr.push(encodeURIComponent(key) + '=' + encodeURIComponent(data[key]))
      }
      //paramArr 转为字符串形式的 encodeData，用&拼接
      encodeData = paramArr.join('&')
    }
    //检查url中有无 ？和检查 ？的位置
    if(method === 'get') {
      const index = url.indexOf('?')
      if(index === -1) {
        url += '?'
      } else if(index < url.length - 1) {
        url += '&'
      }
      url += encodeData
    }
    //open函数用于初始化一个请求，提供请求方式，请求url以及是否执行异步
    xhr.open(method, url, async)
    
    //send方法用于发起请求，传入的参数为要传递的数据
    //get请求不用传递参数或传入null，get请求参数拼接入url中（先用encodeURIComponent进行编码）
    //post请求可以将请求体的参数传入，也需要encodeURIComponent编码
    if(method === 'get') {
      xhr.send(null)
    } else {
      xhr.setRequestHeader('Content-Type','application/x-www-form-urlencoded;charset=UTF-8')
      xhr.send(encodeData)
    } 
    xhr.onprogress = (e) => {
      //已经加载资源大小loaded，资源总大小total
      let precent = e.loaded / e.total 
    }
  })
}
