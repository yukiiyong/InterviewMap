# 格式化Date

将date对象转化为指定的格式
例如：date.Format('yyyy-MM-dd hh:mm:ss')

Date.prototype.Format = function(fmt) {
          let o = {
            'M+': this.getMonth() + 1, //月份
            'd+': this.getDate(), //日
            'h+': this.getHours(), //小时
            'm+': this.getMinutes(), //分
            's+': this.getSeconds(), //秒
            'q+': Math.floor((this.getMonth() + 3)/3), //季度
            'S' : this.getMilliseconds() //毫秒
          }
          if(/(y+)/.test(fmt)) {  //年分开处理
            fmt = fmt.replace(RegExp.$1, (this.getFullYear() + '').substr(4 - RegExp.$1.length))            
          }
          for(let k in o) {
            if(new RegExp('('+ k +')').test(fmt)) {
              fmt = fmt.replace(RegExp.$1, (RegExp.$1.length===1) ? o[k]:('00'+o[k]).substr((''+o[k]).length))
            }
          }
          return fmt
        }

# 生成time与现在之间的时间差

传入一个时间，将其与现在时间做对比，生成传入时间到现在时间的文字表达
例如：let str = friendlyDate(Date.now())

function friendlyDate(time) { //time传入前要进行getTime()
          let date = Date.now()
          console.log(`date: ${date},time:${time}`)
          let c = (date - time) / 1000 //毫秒转秒
          switch(true) {
            case c < 0: 
              return '未来';
            case c < 60:
              return '刚刚';
            case c > 60 && c < 60*60 :
              return Math.floor(c/60)+ '分钟前';
            case c > 60*60 && c < 60*60*24 : //60秒60分12小时 
              return Math.floor(c/60/60) + '小时前';
            case c > 60*60*24 && c < 60*60*24*30 : //天
              return Math.floor(c/60/60/24) + '天前';
            case c > 60*60*24*30 && c < 60*60*24*30*12 : //月
              return Math.floor(c/60/60/24/30) + '月前';
            default :
              return Math.floor(c/60/60/24/30/12) + '年前';
          }
        }
