# 查找字符的个数

# 利用for遍历字符串，将字符的个数存放到一个对象（obj）中，然后返回这个对象

function countNum(str) {
  let obj = []
  let max = 0
  let idx = -1
  for(let i = 0; i < str.length; i++) {
    let key = str[i]
    if(!obj[key]) {
      obj[key] = 1
    }else {
      obj[key]++
    }
    if(obj[key] > max) {
      max = obj[key]
      val = str[i]
    }
  }
  alert(`max:${max},maxValue:${val}`)
  return obj
}

# 利用reduce

维护一个变量max用来存放字符最多的个数，维护一个变量maxStr来存放最多字符数的字符

利用reduce从左到右遍历数组，维护一个对象acc，这个对象的每个属性名为字符，值为字符所在的位置

function countNum(str) {
  let arr = str
  let max = 0
  let maxStr = ''
  if(typeof arr === 'string') {
    arr = arr.split('')
  }
  //利用acc是一个对象，每个属性名是一个数组，存放字符出现的位置
  //利用reduce来遍历数组
  let result = arr.reduce((acc,item,idx) => {
    if(acc[item]) {
      acc[item].push(idx)
    } else {
      acc[item] = [idx]
    }
    if(acc[item].length > max) {
      max = acc[item].length
      maxStr = item
    }
    return acc
  },{})
  alert(`maxStr is ${maxStr},maxNum is ${max}, the location is ${result[maxStr]}`)
  return result
}

# 数组方法

  1. map: 遍历数组，返回回调函数返回值组成的新数组
  2. forEach: 遍历数组;无法break，可以用try/catch中throw new Error停止。 无返回值，可以修改原始数组的值 arr.forEach((item, index, arr) => {})
  3. filter: 过滤
  4. some:有一项为true，则整体为true
  5. every: 有一项为false，则整体为false
  6. join: 通过指定连接符生成字符串
  7. sort(fn) / reverse:排序与反序，改变原数组
  8. concat: 连接数组，不影响原数组，浅挎贝
  9. slice(start，end) 返回截断后的新数组，不改变原数组
  10. splice(start, number, value): 返回删除元素组成的数组，value为插入项，改变原数组
  11. indexOf / lastIndexOf(value, fromIndex)，查找数组项，返回对应的下标
  12. reduce/reduceRight(fn(prev，cur)，defaultPrev) 两两执行，prev为上次化简函数的return值，cur为当前值

# push. pop. shift. unshift 功能及返回值

  1. push: 添加新元素到数组末尾，返回数组长度
  2. pop 移除数组最后一个元素，返回元素本身
  3. shift: 移除数组首个元素，返回元素本身
  4. unshift: 添加新元素到数组开头，返回数组长度
