# 本地仓库初始化

git init

# 项目级别签名

git config user.name [name]

# 系统级别签名

git config --global user.name [name]

# 查看状态

git status 查看工作区，缓存区状态

# 添加操作

git add [文件名] 将工作区新建/修改的内容添加到缓存区

# 提交操作

git commit -m "commit-msg" 将暂存区内容提交到本地库

# 查看历史记录

  • git log
  • git log --pretty=online
  • git reflog

# 前进与后退

  • 基于索引值操作 git reset --hard [哈希索引值]
  • 使用 ^ 符号（一个^只能后退一步）git reset --hard HEAD^
  • 使用～符号（后面跟的数字n为后退的步数） git reset --hard HEAD~2

# 比较文件差异

  • git diff [文件名] 比较工作区和缓存区的文件
  • git diff [本地库历史版本] [文件名]

将工作区文件和本地库历史记录比较，不带文件名是比较多个文件

# 分支管理

优势

  • 可以同时并行多个功能开发，提高效率
  • 各个分支在开发中若某个分支开发失败，不会对其他分支造成影响，失败的分支可以删除后重新开始

# 多人协作

1.git push origin master
2.若有冲突，先git pull 试图合并 (git pull origin master) / (git pull --rebase origin master)
3.若合并有冲突则解决冲突后本地提交（git push）

# 本地新建的分支（非master

若不推送到远程，则对其他人不可见

# 本地创建和远程创建分支

git checkout –b branch-name origin/branch-name

# 建立本地和远程分支联系

git branch -–set-upstream-to branch-name origin/branch-name

# 解除与远程分支联系

git branch --unset-upstrean <branch-name>

# bug分支

  • git stash 把正在工作的分支暂存起来
  • git stash 恢复 git stash apply恢复+ git stash drop删除
  • git stash 恢复 git stash pop 恢复+删除

# 合并删除分支（合并后有历史，非fast-forward）

git merge -–no-ff –m “comment” branch-name

# 删除git缓存区内容

git rm –r –-cached .

# 查找commit-id所包含的分支名

git branch --contains <commit-id>

# 恢复本地被误删的分支（远程拉取）

git checkout -b <branch-name>

# 获取本地仓库发生过的变更

git reflog
git log --pretty=online?获取记录

# 删除远程分支

git push origin --delete <branch-name>
####删除未合并分支
git branch –D branch-name

HEAD指向版本库的头

# 撤销修改

git checkout – filename

  1. 文件自修改后未放入缓存区，则回到和版本库一样的状态
  2. 文件放入缓存区后有修改，则回到添加到缓存区的状态

# 切换分支

git checkout branch-name

# 取消暂存

git reset HEAD <filename>

  1. 撤销（git push 以前，在本地仓库执行）
  2. 文件被修改，未git add
    git checkout filename
    git checkout .
  3. 同时对多个文件执行了git add操作，但本次只想提交一部分文件
    git reset HEAD <filename>(取消缓存)
    文件执行了git add，但想撤销对其的修改
    git reset HEAD filename（取消缓存）
    git checkout filename（撤销修改）
  4. 修改的文件已被git commit，但想再次修改不再产生新的commit
    git commit –amend –m ‘说明’
  5. 在本地执行了多次commit，想撤销到其中某次commit
    git reset [--hard|soft|mixed|merge|keep] [commit|HEAD]

# 回滚 （git push后）

  1. 撤销指定文件到指定版本
    git log <filename> + git checkout <commitId> <filename>

  2. 删除最后一次远程提交
    git revert HAED + git push origin master
    git reset --hard <commit-id>（撤回提交到远程的修改，文件内容恢复到最新提交内容）+ git push origin master -f

  3. revert是放弃指定提交的修改，但会生成一次新的提交，需要填写提交注释，以前的历史记录都在

  4. reset是指将HEAD指针指到指定提交，历史记录中不会出现放弃提交记录

# 查找被意外删除的提交/commit等操作的记录

  1. git fsck --lost-found（适用于add后没commit）
  2. git reflog (适用于commit之后)
    查找出commit-id 后使用 git reset --hard <commit-id> 恢复

# 回滚某次提交

git log + git revert commitID

# 删除某次提交

git rebase –i ‘commitID’
